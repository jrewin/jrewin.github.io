fastify5 MySQL 简单查询 ​

介绍 ​

帮兄弟部门做一个简单的查询服务，我原本是写 java 的，但是这个小需求用 java 真的有点重了。突然想到之前玩过 nodejs ，所以就尝试用 nodejs 实现。他们这个小需求的前端我也帮着写了一个 React + Echart 的几个页面，展示查询结果。不得不说 React + fastify 这一套写下来真的是很方便。

初始化项目 ​

bash
# 初始化package.json
npm init -y
# 安装核心依赖
npm install fastify@5 @fastify/mysql mysql2 dotenv date-fns
# 安装开发依赖（可选，代码格式化）
npm install -D prettier

目录结构 ​

单体极简结构，适配小型查询服务

fastify-mysql-query-api/
├── .env                # 环境配置（数据库、端口、连接池）
├── app.js              # 项目入口（Fastify初始化、插件注册、路由挂载）
├── routes/             # 路由模块（查询API单独抽离，便于维护）
│   └── status.route.js # xxx_status表查询路由
├── services/           # 业务服务层（数据查询、结果处理抽离，解耦路由）
│   └── status.service.js # 数据库操作、结果处理逻辑
└── package.json        # 依赖与启动脚本

创建.env，将敏感配置解耦，部署时仅需修改此文件，核心配置包含 MySQL 连接、连接池（适配频繁查询）、服务端口

bash
# 服务配置
PORT=3600
HOST=0.0.0.0 # 允许外部访问（Windows Server需开放此端口）
NODE_ENV=production # 生产环境

# MySQL配置
MYSQL_HOST=xxx.xx.xx.xx
MYSQL_PORT=3306
MYSQL_USER= root
MYSQL_PASSWORD= password
MYSQL_DATABASE=handler_status # 目标库名
MYSQL_CHARSET=utf8mb4 # 兼容utf8，避免中文乱码

# MySQL连接池配置（关键：适配频繁查询，控制连接数）
MYSQL_CONNECTION_LIMIT=20 # 最大连接数（根据服务器性能调整，Windows Server 2012建议10-30）
MYSQL_QUEUE_LIMIT=0 # 连接池排队无限制
MYSQL_ACQUIRE_TIMEOUT=30000 # 获取连接超时时间（30s）
MYSQL_IDLE_TIMEOUT=600000 # 空闲连接超时（10分钟，释放无用连接）

项目入口（app.js） ​

javascript
require("dotenv").config();
const fastify = require("fastify")({
  logger: {
    level: process.env.NODE_ENV === "production" ? "warn" : "info",
  },
  ignoreTrailingSlash: true, // 忽略路由末尾斜杠（如/api/status/ 和 /api/status 等效）
  bodyLimit: 1024 * 1024, // 限制请求体大小（查询API足够）
});

// 注册MySQL插件（核心：连接池化，适配频繁查询）
fastify.register(require("@fastify/mysql"), {
  promise: true, // 使用Promise风格（替代回调，便于异步处理）
  connectionString: `mysql://${process.env.MYSQL_USER}:${process.env.MYSQL_PASSWORD}@${process.env.MYSQL_HOST}:${process.env.MYSQL_PORT}/${process.env.MYSQL_DATABASE}?charset=${process.env.MYSQL_CHARSET}`,
  pool: {
    connectionLimit: Number(process.env.MYSQL_CONNECTION_LIMIT),
    queueLimit: Number(process.env.MYSQL_QUEUE_LIMIT),
    acquireTimeout: Number(process.env.MYSQL_ACQUIRE_TIMEOUT),
    idleTimeout: Number(process.env.MYSQL_IDLE_TIMEOUT),
  },
});

// 挂载路由（所有/status相关查询路由）
fastify.register(
  require("./routes/status.route"),
  { prefix: "/api" },
  require("@fastify/compress"),
  {
    global: true, // 全局开启压缩
    threshold: 1024, // 数据大于1k时压缩
  },
);

// 全局异常处理
fastify.setErrorHandler((error, request, reply) => {
  request.log.error(error); // 打印错误日志
  reply.status(error.statusCode || 500).send({
    code: error.statusCode || 500,
    message: error.message || "服务器内部错误",
    data: null,
  });
});

// 启动服务
const start = async () => {
  try {
    await fastify.listen({
      port: Number(process.env.PORT),
      host: process.env.HOST,
    });
    console.log(`服务启动成功：http://${process.env.HOST}:${process.env.PORT}`);
  } catch (err) {
    fastify.log.fatal(err);
    process.exit(1);
  }
};

start();

// 捕获进程异常，优雅退出
process.on("uncaughtException", (err) => {
  fastify.log.fatal("未捕获异常：", err);
  process.exit(1);
});
process.on("unhandledRejection", (err) => {
  fastify.log.fatal("未处理的Promise拒绝：", err);
  process.exit(1);
});

业务服务层（services/status.service.js） ​

javascript
const { format, isDate, parseISO } = require("date-fns");

/**
 * 表查询服务
 * @param {Object} fastify - Fastify实例（用于获取MySQL连接池）
 * @param {Object} queryParams - 查询参数：ID、MachineID、Starttime、Endtime
 * @returns {Array} 处理后的查询结果
 */
const queryxxxStatus = async (fastify, queryParams) => {
  const { ID, MachineID, Starttime, Endtime } = queryParams;
  // 1. 初始化查询条件、参数数组（预处理语句用，防SQL注入）
  const whereConditions = [];
  const queryValues = [];

  // 2. 动态拼接查询条件（仅处理有值的参数）
  if (ID && !isNaN(Number(ID))) {
    whereConditions.push("ID = ?");
    queryValues.push(Number(ID)); // 强制转为数字，匹配表结构int类型
  }
  if (MachineID && MachineID.trim()) {
    whereConditions.push("MachineID = ?");
    queryValues.push(MachineID.trim());
  }
  // 处理开始时间：>= Starttime
  if (Starttime) {
    try {
      const startDate = isDate(Starttime) ? Starttime : parseISO(Starttime);
      whereConditions.push("Starttime >= ?");
      queryValues.push(format(startDate, "yyyy-MM-dd HH:mm:ss")); // 统一MySQL时间格式
    } catch (err) {
      throw new Error(
        "Starttime格式错误，需为ISO格式（如2024-01-01T00:00:00）",
      );
    }
  }
  // 处理结束时间：<= Endtime
  if (Endtime) {
    try {
      const endDate = isDate(Endtime) ? Endtime : parseISO(Endtime);
      whereConditions.push("Endtime <= ?");
      queryValues.push(format(endDate, "yyyy-MM-dd HH:mm:ss"));
    } catch (err) {
      throw new Error("Endtime格式错误，需为ISO格式（如2024-01-01T23:59:59）");
    }
  }

  // 3. 拼接SQL语句（查询所有字段，可根据需求精简）
  let sql = "SELECT * FROM xxx_status";
  if (whereConditions.length > 0) {
    sql += ` WHERE ${whereConditions.join(" AND ")}`;
  }
  // 按ID倒序（最新数据在前，符合查询习惯）
  sql += " ORDER BY ID DESC";

  // 4. 执行MySQL查询（从连接池获取连接，自动释放）
  const [rows] = await fastify.mysql.query(sql, queryValues);
  if (rows.length === 0) {
    return [];
  }

  // 5. 处理查询结果（核心：根据业务需求格式化、转换、衍生字段）
  // 示例处理规则（可根据你的需求修改/扩展）：
  const processedData = rows.map((item) => {
    return {
      // 基础字段保留，空值转为''避免前端处理null
      ...item,
      // 数字类型字段：数据库可能返回null，转为0
      Output: item.Output || 0,
      Output_Pass: item.Output_Pass || 0,
      Jam: item.Jam || 0,
      MUBA: item.MUBA || 0,
      // 百分比字段：如Yield/OEE，空值转为'0%'，统一格式
      Yield: item.Yield ? `${item.Yield}`.replace(/[^0-9.]/g, "") + "%" : "0%",
      OEE: item.OEE ? `${item.OEE}`.replace(/[^0-9.]/g, "") + "%" : "0%",
      // 时间字段：转为ISO格式，便于前端时间处理
      Starttime: item.Starttime ? new Date(item.Starttime).toISOString() : null,
      Endtime: item.Endtime ? new Date(item.Endtime).toISOString() : null,
      ReadDate: item.ReadDate ? new Date(item.ReadDate).toISOString() : null,
      // 衍生字段示例：良率=合格数/总数（处理除零异常）
      yieldRate:
        item.Output > 0
          ? ((item.Output_Pass / item.Output) * 100).toFixed(2) + "%"
          : "0%",
      // 衍生字段示例：运行时长（Endtime-Starttime，单位分钟，处理空值）
      runMinutes:
        item.Starttime && item.Endtime
          ? Math.round(
              (new Date(item.Endtime) - new Date(item.Starttime)) / (1000 * 60),
            )
          : 0,
    };
  });

  return processedData;
};

module.exports = { queryxxxStatus };

路由层（routes/status.route.js） ​

javascript
const { queryxxxStatus } = require("../services/status.service");
const moment = require("moment");

/**
 * 挂载xxx_status查询路由
 * @param {Object} fastify - Fastify实例
 * @param {Object} opts - 路由配置
 * @param {Function} done - 回调
 */
module.exports = async (fastify, opts) => {
  // 核心查询接口：/api/status/query
  fastify.get(
    "/status/query",
    {
      // 路由验证（可选，Fastify内置验证，比手动判断更高效）
      schema: {
        querystring: {
          type: "object",
          properties: {
            ID: { type: "integer", minimum: 1 }, // ID为正整数
            MachineID: { type: "string" },
            Starttime: { type: "string" },
            Endtime: { type: "string" },
          },
          additionalProperties: false, // 禁止传递非指定查询参数
        },
        response: {
          // 成功响应格式
          200: {
            type: "object",
            properties: {
              code: { type: "integer", default: 200 },
              message: { type: "string", default: "查询成功" },
              data: { type: "array" },
              total: { type: "integer" }, // 返回数据总数
            },
          },
        },
      },
    },
    async (request, reply) => {
      try {
        // 1. 获取查询参数（Fastify已做schema验证，无需手动判断类型）
        const queryParams = request.query;

        // 自定义时间格式校验与转换
        const parseTime = (timeStr) => {
          const formats = ["YYYY-MM-DD HH:mm:ss", "YYYY-MM-DDTHH:mm:ssZ"]; // 支持多种格式

          const parsed = moment(timeStr, formats, true);
          if (!parsed.isValid()) {
            throw new Error(`无效的时间格式: ${timeStr}`);
          }
          return parsed.toISOString(); // 统一转换为 ISO 格式
        };

        if (queryParams.Starttime) {
          queryParams.Starttime = parseTime(queryParams.Starttime);
        }

        if (queryParams.Endtime) {
          queryParams.Endtime = parseTime(queryParams.Endtime);
        }

        // 2. 调用服务层查询并处理数据
        const data = await queryxxxStatus(fastify, queryParams);
        // 3. 返回标准化响应
        return reply.send({
          code: 200,
          message: "查询成功",
          data,
          total: data.length,
        });
      } catch (err) {
        if (err.message.includes("无效的时间格式")) {
          return reply.status(400).send({
            code: 400,
            message: `时间参数格式不正确，请使用以下格式之一：YYYY-MM-DD HH:mm:ss 或 YYYY-MM-DDTHH:mm:ssZ`,
            data: null,
          });
        }
        // 返回友好的错误信息
        return reply.status(400).send({
          code: 400,
          message: err.message || "参数校验失败",
          data: null,
        });
      }
    },
  );
};

Fastify 服务层优化 ​

开启压缩 ​

注册@fastify/compress插件，压缩响应数据，减少网络传输时间：bash运行

bash
npm install @fastify/compress

在app.js中注册：javascript运行 ​

javascript
fastify.register(require("@fastify/compress"), {
  global: true, // 全局开启压缩
  threshold: 1024, // 数据大于1k时压缩
});

禁用开发模式日志 ​

生产环境logger可调整为warn级别，减少日志 IO 开销，修改app.js的 Fastify 初始化：javascript运行

javascript
const fastify = require("fastify")({
  logger: {
    level: process.env.NODE_ENV === "production" ? "warn" : "info",
  },
  // 其他配置...
});

bash
node app.js
